```python
                '参数字段': ['型号', '变比', '额定一次电流', '额定二次电流'],
                '关键规则': 'TA变比 与 保护装置 匹配',
            },

            # 避雷器
            'YH': {
                '中文名': '避雷器',
                '参数字段': ['型号', '额定电压'],
            },

            # 其他 30+ 类设备
        }

    def 获取符号定义(self, 符号代码: str) -> dict:
        """根据符号代码获取完整定义"""
        return self.符号表.get(符号代码)

    def 校验参数完整性(self, 设备: Device) -> List[str]:
        """校验识别出的设备参数是否完整"""
        符号定义 = self.获取符号定义(设备.符号)
        缺失参数 = []

        for 必需参数 in 符号定义['参数字段']:
            if not getattr(设备, 必需参数, None):
                缺失参数.append(必需参数)

        return 缺失参数

    def 参数合理性检查(self, 设备: Device) -> List[str]:
        """基于知识图谱检查参数合理性"""
        检查结果 = []
        符号定义 = self.获取符号定义(设备.符号)

        for 检查规则 in 符号定义.get('关键规则', []):
            if not self.应用规则(检查规则, 设备):
                检查结果.append(f"⚠️ {检查规则}")

        return 检查结果
```

```python
# sparkflow/model/power_parameter_specs.py

class 电力参数规范:
    """电力工程参数的规范化定义和转换"""

    # 电压等级规范
    电压规范 = {
        '标准形式': {
            '500kV': 500,
            '220kV': 220,
            '110kV': 110,
            '35kV': 35,
            '10kV': 10,
            '0.4kV': 0.4,
        },
        '非标形式': {
            '5级': 500,           # 500kV
            '2.2级': 220,         # 220kV
            '1.1级': 110,         # 110kV
            '35': 35,
            '高压': 110,          # 默认高压
            '中压': 35,           # 默认中压
        },
        '校验规则': [
            ('500kV', '只用于特高压'),
            ('220kV', '通常为区域变电站'),
            ('110kV', '常见主接线'),
            ('35kV', '地级供电'),
            ('10kV', '配电供电'),
        ],
    }

    # 电流单位规范
    电流规范 = {
        '标准单位': 'A',
        '转换规则': {
            'A': lambda x: x,
            'kA': lambda x: x * 1000,
            '千安': lambda x: x * 1000,
            '安培': lambda x: x,
            'mA': lambda x: x / 1000,
        },
        '型号包含电流': {
            'LW36-126': 1260,       # 型号直接包含电流信息
            'LW36-252': 2520,
            'LW8-126': 1260,
        },
    }

    # 设备型号规范
    型号规范 = {
        '断路器': {
            '常见前缀': ['LW', 'SW', 'VB'],
            '型号模式': r'[A-Z]+\d+-\d+(/\d+)?(-\d+)?',  # LW36-126/3150-40
            '校验': lambda x: len(x) >= 5,
        },
        '隔离开关': {
            '常见前缀': ['LW', 'GW', 'HW'],
            '型号模式': r'[A-Z]+\d+-\d+(/\d+)?',
        },
    }

    def 规范化电压(self, 输入值: str) -> float:
        """将各种形式的电压转换为标准 kV 形式"""
        输入值 = 输入值.strip()

        # 尝试标准形式
        if 输入值 in self.电压规范['标准形式']:
            return self.电压规范['标准形式'][输入值]

        # 尝试非标形式
        if 输入值 in self.电压规范['非标形式']:
            return self.电压规范['非标形式'][输入值]

        # 数字直接转换
        try:
            val = float(输入值.replace('kV', '').replace('级', ''))
            if val < 10:  # 一级制表示
                return val * 100
            return val
        except:
            return None

    def 规范化电流(self, 输入值: str) -> float:
        """将各种形式的电流转换为标准 A 形式"""
        # 提取数字和单位
        import re
        match = re.match(r'([\d.]+)\s*([a-zA-Z]*)', 输入值.strip())
        if match:
            数字 = float(match.group(1))
            单位 = match.group(2) or 'A'

            if 单位 in self.电流规范['转换规则']:
                return self.电流规范['转换规则'][单位](数字)

        return None

    def 校验型号格式(self, 设备类型: str, 型号: str) -> bool:
        """校验型号是否符合规范格式"""
        import re
        if 设备类型 in self.型号规范:
            pattern = self.型号规范[设备类型]['型号模式']
            return bool(re.match(pattern, 型号))
        return True  # 未定义则通过

# 使用示例
规范 = 电力参数规范()

# 电压转换
print(规范.规范化电压('1.1级'))          # 输出: 110
print(规范.规范化电压('110kV'))         # 输出: 110

# 电流转换
print(规范.规范化电流('3150A'))         # 输出: 3150
print(规范.规范化电流('3.15kA'))        # 输出: 3150

# 型号校验
print(规范.校验型号格式('断路器', 'LW36-126/3150-40'))  # True
```

#### **模块 3️⃣: 拓扑连接推理引擎**

**目标**: 自动推断设备间的连接关系和上下游依赖

```python
# sparkflow/model/power_topology_reasoning.py

import networkx as nx

class 拓扑连接推理:
    """电气拓扑连接关系推理"""

    def __init__(self):
        # 定义标准的主接线连接规则
        self.连接规则库 = [
            # 规则格式: (前置设备, 后置设备, 连接条件)
            ('母线', '隔离开关', '同间隔'),
            ('隔离开关', '断路器', '同间隔'),
            ('断路器', '电流互感器', '同间隔'),
            ('电流互感器', '母线', '同间隔'),
            ('断路器', '变压器', '相邻设备'),
            ('隔离开关', '母线', '分段'),
        ]

        self.主接线方式 = {
            '双母线': {
                '特征': ['主母线', '旁母线', '切换隔离开关'],
                '连接模式': '母线1 <-> 切换开关 <-> 母线2',
            },
            '单母分段': {
                '特征': ['分段母线', '分段隔离开关'],
                '连接模式': '母线分段1 <-> 分段开关 <-> 母线分段2',
            },
            '桥形': {
                '特征': ['上级母线', '下级母线', '桥接隔离开关'],
                '连接模式': '上母线 <-> 桥开关 <-> 下母线',
            },
        }

        self.拓扑图 = nx.DiGraph()

    def 推理设备连接(self, 识别结果) -> dict:
        """基于识别结果推理设备间的拓扑连接"""

        设备列表 = 识别结果['设备']

        # 步骤 1: 按照连接规则匹配可能的连接
        可能的连接 = []

        for 设备1 in 设备列表:
            for 设备2 in 设备列表:
                if 设备1 == 设备2:
                    continue

                for 规则 in self.连接规则库:
                    前置, 后置, 条件 = 规则

                    if (设备1.类型 == 前置 and
                        设备2.类型 == 后置 and
                        self.满足条件(设备1, 设备2, 条件)):
                        可能的连接.append({
                            '源': 设备1,
                            '目标': 设备2,
                            '规则': 规则,
                            '置信度': self.计算置信度(设备1, 设备2),
                        })

        # 步骤 2: 构建拓扑图
        for 连接 in 可能的连接:
            if 连接['置信度'] > 0.8:  # 置信度阈值
                self.拓扑图.add_edge(
                    连接['源'].id,
                    连接['目标'].id,
                    **连接
                )

        # 步骤 3: 识别主接线方式
        主接线方式 = self.识别主接线方式()

        # 步骤 4: 提取间隔信息
        间隔 = self.提取间隔()

        return {
            '拓扑图': self.拓扑图,
            '连接关系': 可能的连接,
            '主接线方式': 主接线方式,
            '间隔': 间隔,
            '验证报告': self.验证拓扑完整性(),
        }

    def 识别主接线方式(self) -> str:
        """根据拓扑结构识别主接线方式"""

        母线数 = self.统计设备数('母线')
        隔离开关数 = self.统计设备数('隔离开关')

        if 母线数 >= 2 and 隔离开关数 >= 2:
            return '双母线'
        elif 母线数 >= 2 and 隔离开关数 >= 1:
            return '单母分段'
        elif 母线数 >= 2:
            return '桥形'
        else:
            return '单母线'

    def 提取间隔(self) -> List[dict]:
        """从拓扑图中提取逻辑间隔"""

        间隔列表 = []

        # 按连接关系分组（同一条从母线分出的设备为一个间隔）
        # 使用图遍历算法

        visited = set()
        for 节点 in self.拓扑图.nodes():
            if 节点 not in visited:
                子图 = nx.descendants(self.拓扑图, 节点)
                子图.add(节点)
                visited.update(子图)

                间隔列表.append({
                    '间隔号': len(间隔列表) + 1,
                    '设备': list(子图),
                    '主设备': self.找主设备(子图),
                })

        return 间隔列表

    def 验证拓扑完整性(self) -> dict:
        """检查拓扑的完整性和合理性"""

        报告 = {
            '完整性': True,
            '警告': [],
            '错误': [],
        }

        # 检查 1: 是否存在孤立设备
        for 节点 in self.拓扑图.nodes():
            if self.拓扑图.degree(节点) == 0:
                报告['错误'].append(f"孤立设备: {节点}")
                报告['完整性'] = False

        # 检查 2: 是否存在环形连接（错误）
        if not nx.is_directed_acyclic_graph(self.拓扑图):
            报告['警告'].append("检测到循环连接，可能需要检查")

        # 检查 3: 连接关系的置信度
        低置信度连接 = [
            edge for edge in self.拓扫图.edges(data=True)
            if edge[2].get('置信度', 1) < 0.8
        ]
        if 低置信度连接:
            报告['警告'].append(f"存在 {len(低置信度连接)} 个低置信度连接")

        return 报告

# 使用示例
推理引擎 = 拓扑连接推理()

# 推理连接关系

print(f"主接线方式: {结果['主接线方式']}")     # 输出: 双母线
print(f"间隔数量: {len(结果['间隔'])}")         # 输出: 6
```

```python
# sparkflow/rules/power_audit_rules.py

class 电力审图规则库:
    """电力工程专业审图规则"""

    def __init__(self):
        self.规则集 = {
            # 维度 1: 电压等级检查
            '电压_等级合理': {
                '描述': '设备型号的额定电压与图纸电压等级一致',
                '规则': lambda 设备, 图纸电压: 设备.额定电压 == 图纸电压,
                '严重级别': 'ERROR',
            },

            # 维度 2: 电流容量检查
            '电流_容量匹配': {
                '描述': '断路器/隔离开关的开断能力 >= 主接线容量',
                '规则': lambda 设备, 容量: 设备.开断能力 >= 容量,
                '严重级别': 'ERROR',
            },

            # 维度 3: 拓扑完整性检查
            '拓扑_连接完整': {
                '描述': '每个设备都有明确的上游/下游关系',
                '规则': lambda 设备, 拓扑图: 拓扑图.degree(设备) > 0,
                '严重级别': 'ERROR',
            },

            # 维度 4: 参数一致性检查
            '参数_一致性': {
                '描述': '同一间隔的设备参数应协调',
                '规则': lambda 设备1, 设备2: self.参数协调(设备1, 设备2),
                '严重级别': 'WARNING',
            },

            # 维度 5: 符号规范性检查
            '符号_规范性': {
                '描述': '设备符号符合DL/T 5452标准',
                '规则': lambda 符号: 符号 in self.标准符号表,
                '严重级别': 'WARNING',
            },

            # 维度 6: CT变比检查
            'CT_变比适配': {
                '描述': 'CT变比与保护装置等级匹配',
                '规则': lambda ct, 保护: self.检查ct变比(ct, 保护),
                '严重级别': 'WARNING',
            },
        }

    def 执行全面审图(self, 识别结果) -> dict:
        """执行完整的 6+ 维度审图"""

        审图报告 = {
            '总体评分': 100,
            '规则检查结果': [],
            '错误': [],
            '警告': [],
            '建议': [],
        }

        for 规则名, 规则定义 in self.规则集.items():
            try:
                # 应用规则
                检查结果 = self.应用规则(规则名, 规则定义, 识别结果)

                # 记录结果
                审图报告['规则检查结果'].append({
                    '规则': 规则名,
                    '通过': 检查结果['通过'],
                    '详情': 检查结果['详情'],
                    '严重级别': 规则定义['严重级别'],
                })

                # 更新总分
                if not 检查结果['通过']:
                    if 规则定义['严重级别'] == 'ERROR':
                        审图报告['总体评分'] -= 20
                        审图报告['错误'].append(规则名)
                    else:
                        审图报告['总体评分'] -= 5
                        审图报告['警告'].append(规则名)

            except Exception as e:
                审图报告['建议'].append(f"规则 {规则名} 执行异常: {e}")

        return 审图报告

# 使用示例

print(f"警告数: {len(报告['警告'])}")

# 详细输出
for 检查 in 报告['规则检查结果']:
    状态 = "✓" if 检查['通过'] else "✗"
    print(f"{状态} {检查['规则']}: {检查['详情']}")
```
